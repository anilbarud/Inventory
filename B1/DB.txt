CREATE TABLE SUPPLIER(
KDSUPPLIER CHAR(15),
NMSUPPLIER VARCHAR(45),
ALAMAT TEXT,
TLP VARCHAR(25),
FAX VARCHAR(25),
PRIMARY KEY(KDSUPPLIER)
);

CREATE TABLE CUSTOMER(
KDCUSTOMER CHAR(15),
NMCUSTOMER VARCHAR(45),
ALAMAT TEXT,
TLP VARCHAR(25),
PRIMARY KEY(KDCUSTOMER)
);

CREATE TABLE BARANG(
KDBARANG CHAR(15),
NMBARANG VARCHAR(45),
JENIS VARCHAR(35),
HARGA NUMERIC(9),
STOK NUMERIC(9),
PRIMARY KEY(KDBARANG)
);

CREATE TABLE PEMBELIAN(
NOBELI CHAR(15),
KDSUPPLIER CHAR(15),
TANGGAL DATETIME,
PRIMARY KEY(NOBELI),
FOREIGN KEY(KDSUPPLIER) REFERENCES SUPPLIER(KDSUPPLIER)
);

CREATE TABLE PEMBELIAN_DET(
NOBELI CHAR(15),
KDBARANG CHAR(15),
JUMLAH NUMERIC(9),
TOTAL NUMERIC(9),
PRIMARY KEY(NOBELI,KDBARANG),
FOREIGN KEY(NOBELI) REFERENCES PEMBELIAN(NOBELI),
FOREIGN KEY(KDBARANG) REFERENCES BARANG(KDBARANG)
);

CREATE TABLE PENJUALAN(
NOJUAL CHAR(15),
KDCUSTOMER CHAR(15),
TANGGAL DATETIME,
PRIMARY KEY(NOJUAL),
FOREIGN KEY(KDCUSTOMER) REFERENCES CUSTOMER(KDCUSTOMER)
);

CREATE TABLE PENJUALAN_DET(
NOJUAL CHAR(15),
KDBARANG CHAR(15),
JUMLAH NUMERIC(9),
TOTAL NUMERIC(9),
PRIMARY KEY(NOJUAL,KDBARANG),
FOREIGN KEY(NOJUAL) REFERENCES PENJUALAN(NOJUAL),
FOREIGN KEY(KDBARANG) REFERENCES BARANG(KDBARANG)
);
