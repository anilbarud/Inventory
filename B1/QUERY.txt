CREATE TABLE SUPPLIER(
KDSUPPLIER CHAR(10),
NMSUPPLIER VARCHAR(35),
ALAMAT TEXT,
TLP VARCHAR(25),
FAX VARCHAR(25),
PRIMARY KEY(KDSUPPLIER)
);

CREATE TABLE CUSTOMER(
KDCUSTOMER CHAR(10),
NMCUSTOMER VARCHAR(35),
ALAMAT TEXT,
TLP VARCHAR(25),
PRIMARY KEY(KDCUSTOMER)
);

CREATE TABLE BARANG(
KDBARANG CHAR(10),
NMBARANG VARCHAR(50),
HARGA NUMERIC(9),
STOK NUMERIC(9),
PRIMARY KEY(KDBARANG)
);

CREATE TABLE PEMBELIAN(
NOBELI CHAR(10),
KDSUPPLIER CHAR(10),
TGL DATETIME,
PRIMARY KEY(NOBELI),
FOREIGN KEY(KDSUPPLIER) REFERENCES SUPPLIER(KDSUPPLIER)
);

CREATE TABLE PEMBELIAN_DET(
NOBELI CHAR(10),
KDBARANG CHAR(10),
JUMLAH NUMERIC(9),
TOTAL NUMERIC(9),
PRIMARY KEY(NOBELI,KDBARANG),
FOREIGN KEY(NOBELI) REFERENCES PEMBELIAN(NOBELI),
FOREIGN KEY(KDBARANG) REFERENCES BARANG(KDBARANG)
);

CREATE TABLE PENJUALAN(
NOJUAL CHAR(10),
KDCUSTOMER CHAR(10),
TGL DATETIME,
PRIMARY KEY(NOJUAL),
FOREIGN KEY(KDCUSTOMER) REFERENCES CUSTOMER(KDCUSTOMER)
);

CREATE TABLE PENJUALAN_DET(
NOJUAL CHAR(10),
KDBARANG CHAR(10),
JUMLAH NUMERIC(9),
TOTAL NUMERIC(9),
PRIMARY KEY(NOJUAL,KDBARANG),
FOREIGN KEY(NOJUAL) REFERENCES PENJUALAN(NOJUAL),
FOREIGN KEY(KDBARANG) REFERENCES BARANG(KDBARANG)
);
